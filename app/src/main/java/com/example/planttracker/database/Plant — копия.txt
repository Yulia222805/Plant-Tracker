package com.example.planttracker.database

import androidx.room.*

@Entity(tableName = "plants")
data class Plant(
    @PrimaryKey(autoGenerate = true) val id: Long = 0,

    @ColumnInfo(name = "name") val name: String,
    @ColumnInfo(name = "description") val description: String? = null,

    // Полив
    @ColumnInfo(name = "watering_interval_days") val wateringIntervalDays: Int = 7,
    @ColumnInfo(name = "last_watered") val lastWatered: Long = System.currentTimeMillis(),

    // Удобрение
    @ColumnInfo(name = "fertilizing_interval_days") val fertilizingIntervalDays: Int = 30,
    @ColumnInfo(name = "last_fertilized") val lastFertilized: Long = System.currentTimeMillis(),

    // Фото (позже)
    @ColumnInfo(name = "image_path") val imagePath: String? = null
)
